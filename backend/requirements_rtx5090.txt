#
#     Option B -- CUDA 12.8 (newer, better Blackwell support):

#         --index-url https://download.pytorch.org/whl/cu128
#
#   Step 2: pip install -r requirements_rtx5090.txt
#   Step 3: pip install -r requirements_voice.txt
#   Step 4: pip install -r services/TripoSR/requirements.txt
